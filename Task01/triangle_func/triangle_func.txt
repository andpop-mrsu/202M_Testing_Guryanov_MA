# Позитивные кейсы
1. a = 3, b = 4, c = 5 → Ожидаемый результат: 'nonequilateral'
2. a = 5, b = 5, c = 5 → Ожидаемый результат: 'equilateral'
3. a = 5, b = 5, c = 8 → Ожидаемый результат: 'isosceles'

# Негативные кейсы
1. a = -3, b = 4, c = 5 → Ожидаемый результат: Ошибка (длины сторон должны быть положительными)
2. a = 1, b = 1, c = 3 → Ожидаемый результат: Ошибка (не образует треугольник)
3. a = 0, b = 4, c = 5 → Ожидаемый результат: Ошибка (длины сторон должны быть положительными)
